def user_poke_edit(request, user_id, fish_id):
    user = get_object_or_404(User, id=user_id)
    fish = get_object_or_404(Fish, id=fish_id)

    # Verificar si ya existe una relación entre el usuario y el pez
    user_fish = UserFish.objects.filter(user=user, fish=fish).first()

    if request.method == 'POST':
        # Actualizar o crear la relación del usuario con el pez
        captured = request.POST.get('captured', 'false') == 'true'
        if user_fish:
            user_fish.captured = captured
            user_fish.save()
        else:
            UserFish.objects.create(user=user, fish=fish, captured=captured)

        return redirect('fish_detail', user_id=user.id)  # Redirigir a la página de detalles o lista

    return render(
        request,
        "user_poke_edit.html",
        {
            "user": user,
            "fish": fish,
            "captured": user_fish.captured if user_fish else False,
        },
    )

document.addEventListener('DOMContentLoaded', function() {
    const userPic = document.getElementById("user-pic");
    const stickyElement = document.querySelector(".info-headers");
    let isSticky = false; // Variable para controlar el estado de la fijación

    // Función para comprobar si el usuario está haciendo scroll y el sticky debe activarse
    function checkSticky() {
        const scrollTop = window.scrollY; // Posición del scroll
        const stickyOffset = stickyElement.offsetTop; // Cuando el sticky element llega al tope

        if (scrollTop >= stickyOffset && !isSticky) {
            // Cuando el scroll supera la posición del sticky element, activamos el sticky
            userPic.classList.add("fixed");
            userPic.classList.remove("default");
            isSticky = true; // Marcamos que ya está en estado fijo
        } else if (scrollTop < stickyOffset && isSticky) {
            // Cuando el scroll vuelve arriba y el sticky estaba activo, lo desactivamos
            userPic.classList.add("default");
            userPic.classList.remove("fixed");
            isSticky = false; // Marcamos que ya no está en estado fijo
        }
    }

    // Escuchamos el evento de scroll para aplicar la lógica de sticky
    window.addEventListener('scroll', checkSticky);

    // Inicializamos el estado al cargar la página
    checkSticky();
});
