def user_poke_edit(request, user_id, fish_id):
    user = get_object_or_404(User, id=user_id)
    fish = get_object_or_404(Fish, id=fish_id)

    # Verificar si ya existe una relación entre el usuario y el pez
    user_fish = UserFish.objects.filter(user=user, fish=fish).first()

    if request.method == 'POST':
        # Actualizar o crear la relación del usuario con el pez
        captured = request.POST.get('captured', 'false') == 'true'
        if user_fish:
            user_fish.captured = captured
            user_fish.save()
        else:
            UserFish.objects.create(user=user, fish=fish, captured=captured)

        return redirect('fish_detail', user_id=user.id)  # Redirigir a la página de detalles o lista

    return render(
        request,
        "user_poke_edit.html",
        {
            "user": user,
            "fish": fish,
            "captured": user_fish.captured if user_fish else False,
        },
    )
